import { getCurrentLanguage } from "@/lib/dictionary";
import ProjectPostLayout from "@/components/projects/project-post-layout";
import { getPageMetadata } from "@/config/metadata";

export async function generateMetadata() {
    const lang = await getCurrentLanguage();
    return getPageMetadata(lang, {
        title: projectData.title,
        description: projectData.description,
        image: projectData.image,
    });
}

export default async function Layout(props) {
    const lang = await getCurrentLanguage();
    return <ProjectPostLayout project={projectData} lang={lang}>{props.children}</ProjectPostLayout>;
}
