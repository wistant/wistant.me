import CertLayout from "@/components/certifications/cert-layout";

import { getPageMetadata } from "@/config/metadata";

export async function generateMetadata(props) {
  const params = await props.params;
  const { lang } = params;
  return getPageMetadata(lang, {
    title: certData.title,
    description: `Certification: ${certData.title} from ${certData.issuer}`,
    image: certData.image,
  });
}

export const certData = {
  title: "AWS Certified DevOps Engineer – Professional",
  date: "2024-05-20",
  image: "/certifications/aws-devops-professional.png",
  issuer: "Amazon Web Services (AWS)",
  slug: "aws-devops-professional",
};
